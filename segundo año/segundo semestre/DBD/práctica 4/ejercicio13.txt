Club = (IdClub, nombreClub, ciudad)
Complejo = (IdComplejo, nombreComplejo, IdClub(fk))
Cancha = (IdCancha, nombreCancha, IdComplejo(fk))
Entrenador = (IdEntrenador, nombreEntrenador, fechaNacimiento, direccion)
Entrenamiento = (IdEntrenamiento, fecha, IdEntrenador(fk), IdCancha(fk))


1. Listar nombre, fecha de nacimiento y dirección de entrenadores que hayan tenido
entrenamientos durante 2023.


SELECT e.nombreEntrenador, e.fechaNacimiento, e.direccion
FROM Entrenador e
WHERE EXISTS (
    SELECT *
    FROM Entrenamiento x
    WHERE e.IdEntrenador = x.IdEntrenador
      AND YEAR(x.fecha) = 2023
);


2. Listar para cada cancha del complejo “Complejo 1”, la cantidad de entrenamientos que se
realizaron durante el 2022. Informar nombre de la cancha y cantidad de entrenamientos.


SELECT can.nombreCancha, COUNT(x.IdEntrenamiento) AS cantidad_entrenamientos
FROM Cancha can
INNER JOIN Complejo com ON can.IdComplejo = com.IdComplejo
LEFT JOIN Entrenamiento x ON can.IdCancha = x.IdCancha AND YEAR(x.fecha) = 2022
WHERE com.nombreComplejo = 'Complejo 1'
GROUP BY can.IdCancha, can.nombreCancha;


3. Listar los complejos donde haya realizado entrenamientos el entrenador “Jorge Gonzalez”.
Informar nombre de complejo, ordenar el resultado de manera ascendente.


SELECT DISTINCT com.nombreComplejo
FROM Complejo com
WHERE EXISTS (
    SELECT *
    FROM Cancha c
    INNER JOIN Entrenamiento x ON c.IdCancha = x.IdCancha
    INNER JOIN Entrenador e ON x.IdEntrenador = e.IdEntrenador
    WHERE c.IdComplejo = com.IdComplejo
      AND e.nombreEntrenador = 'Jorge Gonzalez'
)
ORDER BY com.nombreComplejo ASC;



4. Listar nombre, fecha de nacimiento y dirección de entrenadores que hayan entrenado en los
clubes con nombre “Everton” y “Estrella de Berisso”.


SELECT DISTINCT e.nombreEntrenador, e.fechaNacimiento, e.direccion
FROM Entrenador e
WHERE EXISTS ( 
  SELECT *
  FROM Entrenamiento x
  INNER JOIN Cancha can ON x.IdCancha = can.IdCancha
  INNER JOIN Complejo com ON can.IdComplejo= com.IdComplejo
  INNER JOIN Club club ON com.IdClub=club.IdClub
  WHERE x.IdEntrenador=e.IdEntrenador AND club.nombreClub='Everton'
)
AND EXISTS ( 
  SELECT *
  FROM Entrenamiento x
  INNER JOIN Cancha can ON x.IdCancha = can.IdCancha
  INNER JOIN Complejo com ON can.IdComplejo= com.IdComplejo
  INNER JOIN Club club ON com.IdClub=club.IdClub
  WHERE x.IdEntrenador=e.IdEntrenador AND club.nombreClub='Estrella de Berisso'
);


5. Listar todos los clubes en los que entrena el entrenador “Marcos Perez”. Informar nombre del
club y ciudad.


SELECT DISTINCT club.nombreClub, club.ciudad
FROM Club club
INNER JOIN Complejo com ON club.IdClub=com.IdClub
INNER JOIN Cancha can ON com.IdComplejo=can.IdComplejo
INNER JOIN Entrenamiento x ON can.IdCancha=x.IdCancha
INNER JOIN Entrenador e ON x.IdEntrenador=e.IdEntrenador
WHERE e.nombreEntrenador = 'Marcos Perez'


6. Eliminar los entrenamientos del entrenador ‘Juan Perez’.

DELETE FROM Entrenamiento WHERE IdEntrenador = (SELECT IdEntrenador FROM Entrenador WHERE nombreEntrenador = 'Juan Perez')
